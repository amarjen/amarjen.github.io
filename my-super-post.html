# Datos para generar el mapa de estabilidad financiera mundial.
# Fuente: Publicación semestral GFSR del FMI 

# Etiquetas
riesgos = ['Riesgos de mercados emergentes',
           'Riesgos de crédito',
           'Riesgos de mercado y liquidez',
           'Apetito de riesgo',
           'Condiciones monetarias y financieras',
           'Riesgos macroeconómicos']

# Índice de la serie temporal
ix = ['1/4/2007','1/10/2007','1/4/2008','1/10/2008','1/4/2009','1/10/2009','1/4/2010','1/10/2010',
      '1/4/2011','1/10/2011','1/4/2012','1/10/2012','1/4/2013','1/10/2013']
ix = [pd.to_datetime(d, dayfirst=True) for d in ix]

# Datos
em = [3.0 ,3.0 ,5.0 ,6.0 , 8.9 ,6.9 ,5.9 ,5.0 ,4.0 ,5.0 ,5.0 ,6.0 ,5.0 ,6.0 ] # Riesgos de mercados emergentes
ml = [4.0 ,6.0 ,7.0 ,9.0 , 9.0 ,7.0 ,6.0 ,6.0 ,6.0 ,6.9 ,6.0 ,5.9 ,4.0 ,5.0 ] # Riesgos de mercado y liquidez
mr = [3.0 ,4.0 ,7.0 ,8.0 , 9.0 ,8.0 ,7.0 ,7.9 ,6.1 ,7.0 ,7.0 ,8.0 ,7.0 ,7.0 ] # Riesgos macroeconómicos
mf = [7.0 ,6.0 ,5.0 ,4.0 , 3.0 ,4.1 ,5.1, 4.1 ,6.1 ,6.0 ,6.0 ,6.0 ,6.9 ,6.9 ] # Condiciones monetarias
cr = [2.0 ,5.0 ,8.0 ,9.0 ,10.0 ,9.0 ,8.0 ,8.0 ,7.0 ,8.1 ,8.1 ,8.1 ,7.1 ,7.1 ] # Riesgo de crédito
ra = [7.0 ,5.0 ,3.0 ,2.0 , 1.0 ,4.0 ,5.0, 4.0 ,6.0 ,4.0 ,5.0 ,4.0 ,6.9 ,6.0 ] # Apetito al riesgo

# Representación de la serie temporal en formato 'DataFrame' de la libreria 'Pandas'
fin_map = pd.DataFrame(dict(zip(riesgos, [em, cr, ml, mf, ra, mr])), columns=riesgos, index=ix)
